81    /**
82     * Checks whether it is run in a Liferay Portal environment, regardless of user authentication status.
83     * 
84     * @return true, if run in a Liferay Portal environement, regardless of user authentication status.
85     */
86    public static boolean isLiferayPortlet() {
87      VaadinSession.getCurrent().getService();
88      Object PORTLET_ID =
89          VaadinService.getCurrentRequest().getAttribute(
90              PortalUtils.LiferaySpecificAttribute);
91      if (LOG.isInfoEnabled()) {
92        LOG.info("PORTLET_ID = {}", PORTLET_ID);
93        final Enumeration<String> attributeNames = VaadinService.getCurrentRequest().getAttributeNames();
94        while (attributeNames.hasMoreElements()) {
95          final String attributeName = attributeNames.nextElement();
96          LOG.info("  {} = {}", attributeName, VaadinService.getCurrentRequest().getAttribute(attributeName));
97        }
98      }
99      return PORTLET_ID != null;
100   }

102   /**
103 	 * Transforms bytes into human readable byte count eg.  110592:   110.6 kB (si = true)  108.0 KiB (si =false)
104 	 * @param bytes
105 	 * @param si
106 	 * @return bytes in readable count and unit
107 	 */
108 	public static String humanReadableByteCount(long bytes, boolean si) {
109 	    int unit = si ? 1000 : 1024;
110 	    if (bytes < unit) return bytes + " B";
111 	    int exp = (int) (Math.log(bytes) / Math.log(unit));
112 	    String pre = (si ? "kMGTPE" : "KMGTPE").charAt(exp-1) + (si ? "" : "i");
113 	    return String.format("%.1f %sB", bytes / Math.pow(unit, exp), pre);
114 	}

117     /**
118      * Generates path to image path e.g. https://portal-testing.qbic.uni-tuebingen.de/slideshow-portlet/VAADIN/
119      * @param subfolder needs to be a folder placed in VAADIN folder
120      * @return path to the pictures folder of a portlet
121      */
122     public static String buildImagePath(String subfolder) {
123         StringBuilder pathBuilder = new StringBuilder();
124 
125         if (isLiferayPortlet()) {
126             Properties prop = new Properties();
127             InputStream in = PortalUtils.class.getClassLoader()
128                     .getResourceAsStream("WEB-INF/liferay-plugin-package.properties");
129             try {
130                 prop.load(in);
131                 in.close();
132             } catch (IOException e1) {
133                 e1.printStackTrace();
134             }
135             String portletName = prop.getProperty("name");
136 
137 
138             //pathBuilder.append("https://");
139             URI location = UI.getCurrent().getPage().getLocation();
140             // http
141             pathBuilder.append(location.getScheme());
142             pathBuilder.append("://");
143             // host+port
144             pathBuilder.append(location.getAuthority());
145 
146             String port = (Integer.toString(location.getPort()));
147             if (location.toString().contains(port)) {
148                 pathBuilder.append(":");
149                 pathBuilder.append(port);
150             }
151             // pathBuilder.append("portal-testing.qbic.uni-tuebingen.de");
152             pathBuilder.append("/");
153             pathBuilder.append(portletName);
154         }
155 
156         //path to images folder
157         pathBuilder.append("/VAADIN/"+subfolder);
158         LOG.info(pathBuilder.toString());
159         return pathBuilder.toString();
160     }

170   /**
171    * Builds and returns a {@link com.vaadin.ui.Layout} displaying a "please log in" message. Portlets using this method must include the mail9.png and
172    * lock12.png image resources (both sized 64x64 pixels) from the {@code src/main/resources/img} folder in this project to the corresponding folder in the
173    * portlet project (usually {@code src/main/webapp/VAADIN/themes/mytheme}).
174    *
175    * @return a {@link com.vaadin.ui.Layout} informing users to log in.
176    */
177   public static Layout buildNotLoggedInLayout() {
178     // Mail to qbic
179     ExternalResource resource = new ExternalResource("mailto:support@qbic.zendesk.com");
180     Link mailToQbicLink = new Link("", resource);
181     mailToQbicLink.setIcon(new ThemeResource("mail9.png"));
182 
183 
184     ThemeDisplay themedisplay =
185         (ThemeDisplay) VaadinService.getCurrentRequest().getAttribute(WebKeys.THEME_DISPLAY);
186 
187     // redirect to liferay login page
188     Link loginPortalLink = new Link("", new ExternalResource(themedisplay.getURLSignIn()));
189     loginPortalLink.setIcon(new ThemeResource("lock12.png"));
190 
191     // left part of the page
192     VerticalLayout signIn = new VerticalLayout();
193     signIn.addComponent(new Label("<h3>Sign in to manage your projects and access your data:</h3>",
194         ContentMode.HTML));
195     signIn.addComponent(loginPortalLink);
196     signIn.setStyleName("no-user-login");
197     // right part of the page
198     VerticalLayout contact = new VerticalLayout();
199     contact.addComponent(new Label(
200         "<h3>If you are interested in doing projects get in contact:</h3>", ContentMode.HTML));
201     contact.addComponent(mailToQbicLink);
202     contact.setStyleName("no-user-login");
203 
204     // build final layout, with some gaps between
205     HorizontalLayout notSignedInLayout = new HorizontalLayout();
206     Label expandingGap1 = new Label();
207     expandingGap1.setWidth("100%");
208     notSignedInLayout.addComponent(expandingGap1);
209     notSignedInLayout.addComponent(signIn);
210 
211     notSignedInLayout.addComponent(contact);
212     notSignedInLayout.setExpandRatio(expandingGap1, 0.16f);
213     notSignedInLayout.setExpandRatio(signIn, 0.36f);
214 
215     notSignedInLayout.setExpandRatio(contact, 0.36f);
216 
217     notSignedInLayout.setWidth("100%");
218     notSignedInLayout.setSpacing(true);
219 
220     return notSignedInLayout;
221   }
