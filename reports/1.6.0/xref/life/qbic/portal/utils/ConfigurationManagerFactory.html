11  /**
12   * The ConfigurationManager Factory has only the {@link #getInstance()} method. This class maintains a static reference to an implementation of the
13   * {@link ConfigurationManager}, as a sort of "proxy singleton" (developer's quotes). Please note that the term "proxy singleton" does not probably exist, but
14   * it is not easy to describe this pattern.
15   *
16   * This odd design choice (having a factory that acts as a singleton) was chosen due to the fact that there are hard references to this class all over our
17   * portlets, so changing the signature of the methods required to get an instance of a {@link ConfigurationManager}, would mean that no portlet would compile
18   * if pure singletons (without a factory) are used. Such is life, I guess.
19   *
20   * Depending on whether the portlet runs on its own or in a Portal environment a different configuration file will be used to initialize the instance of
21   * {@llink ConfigurationManager}.
22   * 
23   * @author wojnar, chahuistle
24   * 
25   */
26  public class ConfigurationManagerFactory {
27  
28    private static final Logger LOG = LogManager.getLogger(ConfigurationManagerFactory.class);
29  
30    // this is a "proxy singleton" (developer's quotes, I actually don't know how to name this "pattern", again, my quotes)
31    private static ConfigurationManager PROXY_INSTANCE;
32    /**
33     * Returns the (singleton) instance of a {@link ConfigurationManager} that has been configured with the right configuration file.
34     * 
35     * @return Instance of {@link ConfigurationManager}.
36     */
37    public static synchronized ConfigurationManager getInstance() {
38      if (PROXY_INSTANCE == null) {
39        final Properties properties;
40        // find out how we will populate the Properties object, that is, whether from a Liferay config file or from a file in the classpath
41        if (PortalUtils.isLiferayPortlet()) {
42          LOG.info("Running on a Liferay portal");
43          // this baby is running on a portal, how exciting!
44          properties = PortletProps.getProperties();
45        } else {
46          // load from a properties file that is in the classpath
47          LOG.info("It seems that this is not a Liferay instance, reading local configuration file");
48          properties = new Properties();
49            try (final InputStream inputStream = ConfigurationManagerFactory.class.getClassLoader()
50                .getResourceAsStream(QBiCPortletUI.DEVELOPER_PROPERTIES_FILE_PATH)) {
51              if (inputStream == null) {
52                LOG.warn("Your local configuration file was not found in the classpath. This might not be a problem. Perhaps you forgot to add {} in the classpath?", QBiCPortletUI.DEVELOPER_PROPERTIES_FILE_PATH);
53              } else {
54                properties.load(inputStream);
55              }
56            } catch (IOException e) {
57              throw new RuntimeException("Could not load local configuration file " + QBiCPortletUI.DEVELOPER_PROPERTIES_FILE_PATH, e);
58            }
59        }
60        PROXY_INSTANCE = new PropertiesBasedConfigurationManager(properties);
61      }
62      return PROXY_INSTANCE;
63    }
64  }
