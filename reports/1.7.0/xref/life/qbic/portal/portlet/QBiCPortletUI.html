17  
18  /**
19   * This is the class from which all other QBiC portlets derive.
20   */
21  @Theme("mytheme")
22  @SuppressWarnings("serial")
23  @Widgetset("life.qbic.portlet.AppWidgetSet")
24  public abstract class QBiCPortletUI extends UI {
25  
26    /**
27     * Location of the properties file where maven stores version/repository url.
28     */
29    public static final String PORTLET_PROPERTIES_FILE_PATH = "portlet.properties";
30  
31    /**
32     * Location of the local configuration. This is useful when testing portlets loocally using {@code mvn jetty:run}.
33     */
34    public static final String DEVELOPER_PROPERTIES_FILE_PATH = "developer.properties";
35  
36    static final String INFO_LABEL_CLASS = "portlet-footer";
37    static final String INFO_LABEL_ID = "qbic-portlet-info-label";
38    static final String DEFAULT_REPO = "http://github.com/qbicsoftware";
39    static final String DEFAULT_VERSION = "0.0.1-alpha";
40  
41  
42    private static final Logger LOG = LogManager.getLogger(QBiCPortletUI.class);
43  
44    private final String portletVersion;
45    private final String portletRepoURL;
46  
47    /**
48     * Default constructor.
49     */
50    public QBiCPortletUI() {
51      // load url/version from portlet.properties
52      LOG.info("Initializing QBiCPortletUI");
53      final Properties portletProperties = new Properties();
54      try (final InputStream propertiesFileStream =
55          QBiCPortletUI.class.getClassLoader().getResourceAsStream(PORTLET_PROPERTIES_FILE_PATH)) {
56        portletProperties.load(propertiesFileStream);
57        if (!portletProperties.containsKey("version")
58            || !portletProperties.containsKey("repository.url")) {
59          LOG.error("Missing version and/or repository url properties in portlet.properties file. "
60              + "The file should contain the 'version' and 'repository.url' properties. "
61              + "Using default version and/or repository url.");
62        }
63      } catch (final Exception e) {
64        LOG.error("Error loading portlet.properties. Portlet will display bogus version/url values. "
65            + "Check that portlet.properties is found under the /WEB-INF/classes folder (this file "
66            + "is copied from src/main/resources/portlet.properties during the build).", e);
67      }
68  
69      portletVersion =
70          StringUtils.trimToEmpty(portletProperties.getProperty("version", DEFAULT_VERSION));
71      portletRepoURL =
72          StringUtils.trimToEmpty(portletProperties.getProperty("repository.url", DEFAULT_REPO));
73    }
74  
75    @Override
76    protected final void init(final VaadinRequest request) {
77      LOG.info("Version " + portletVersion);
78      final VerticalLayout layout = new VerticalLayout();
79      layout.setMargin(false);
80      // add the portlet
81      layout.addComponent(getPortletContent(request));
82  
83      addPortletInfo(layout);
84      setContent(layout);
85    }
86  
87    @Override
88    public final void setContent(final Component content) {
89      super.setContent(content);
90    }
91  
92    // adds version and repository information to the passed layout
93    private void addPortletInfo(final Layout layout) {
94      final Label portletInfoLabel = new Label("version " + portletVersion + " (<a href=\""
95          + portletRepoURL + "\">" + portletRepoURL + "</a>)", ContentMode.HTML);
96      portletInfoLabel.addStyleName(INFO_LABEL_CLASS);
97      portletInfoLabel.setId(INFO_LABEL_ID);
98      layout.addComponent(portletInfoLabel);
99    }
100 
101   /**
102    * Provide the content that will be displayed.
103    */
104   protected abstract Layout getPortletContent(final VaadinRequest request);
105 }
